## How each base is used

<Table
  headers={["Form", "Base + Kana", "Example"]}
  data={[
    ["Negative", "A-base + ない", "書かない — kakanai"],
    ["Negative past", "A-base + なかった", "書かなかった — kakanakatta"],
    ["Prohibition (casual)", "A-base + ないで", "書かないで — don't write"],

    ["Polite", "I-base + ます", "書きます — kakimasu"],
    ["Want to (desire)", "I-base + たい", "書きたい — kakitai"],
    ["Easy to do", "I-base + やすい", "書きやすい — easy to write"],
    ["Hard to do", "I-base + にくい", "書きにくい — hard to write"],
    ["And / connecting verbs", "I-base + ...", "書き、読んで… — write and..."],

    ["Dictionary form", "U-base", "書く — kaku"],
    ["Casual volitional alt", "U-base +（う form change）", "書く → 書こう — let's write"],

    ["Command!", "E-base + ！", "書け！ — kake!"],
    ["Potential (some verbs)", "E-base + る", "書ける — kakeru (can write)"],

    ["Let's / volitional", "O-base + う", "書こう — kakou"],
    ["Intention / shall I?", "O-base + うか", "書こうか — shall we write?"],

    ["Causative (intro)", "A-base + せる （N5/N4+）", "書かせる — make/let (someone) write"],

]}
/>
